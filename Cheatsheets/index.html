HTML Cheatsheet

HyperText Markup Language – Die Struktur jeder Webseite

Grundgerüst

<!DOCTYPE html>
<html lang="de">
  <head>
    <meta charset="UTF-8" />
    <meta name="viewport"
      content="width=device-width,
      initial-scale=1.0" />
    <title>Meine Seite</title>
    <link rel="stylesheet"
      href="style.css" />
  </head>
  <body>
    <!-- Inhalt hier -->
    <script src="script.js">
    </script>
  </body>
</html>
Jede HTML-Datei beginnt mit diesem Grundgerüst.

Text-Elemente

<h1> bis <h6>Überschriften (h1 = größte)
<p>Absatz (Paragraph)
<br />Zeilenumbruch
<strong>Fett (wichtig)
<em>Kursiv (betont)
<span>Inline-Container (kein eigener Stil)
<blockquote>Zitat-Block

Links & Bilder

<a href="https://example.com"

     width="300" />

Listen

  <li>Punkt 1</li>
  <li>Punkt 2</li>
</ul>

<!-- Geordnete Liste -->

CSS Cheatsheet

Cascading Style Sheets – Das Aussehen deiner Webseite

Selektoren

/* Element */
p { color: blue; }

/* Klasse */
.card { background: white; }

/* ID */
#header { height: 60px; }

/* Verschachtelt */
nav a { color: red; }

/* Direkt Kind */
ul > li { margin: 5px; }

/* Mehrere Elemente */
h1, h2, h3 { font-weight: bold; }

Box-Modell

margin
border
padding
content
.box {
  margin: 20px;      /* Außenabstand */
  border: 1px solid; /* Rahmen */
  padding: 15px;     /* Innenabstand */
  width: 200px;      /* Breite */

  /* Wichtig! */
  box-sizing: border-box;
}

Display-Typen

blockVolle Breite, neue Zeile
inlineNur so breit wie Inhalt
inline-blockInline + Breite/Höhe
flexFlexbox-Container
gridGrid-Container
noneElement verstecken

Flexbox

.container {
  display: flex;

  /* Hauptachse */
  justify-content: center;
  /* flex-start | flex-end |
     center | space-between |
     space-around */

  /* Querachse */
  align-items: center;
  /* flex-start | flex-end |
     center | stretch */

  /* Richtung */
  flex-direction: row;
  /* row | column |
     row-reverse | column-reverse */

  /* Umbruch */
  flex-wrap: wrap;

  /* Abstand */
  gap: 1rem;
}

Positionierung

staticNormal (Standard)
relativeRelativ zur Normalposition
absoluteRelativ zum nächsten rel. Elternteil
fixedFixiert im Viewport
stickyKlebt beim Scrollen
.fixed-header {
  position: fixed;
  top: 0;
  left: 0;
  width: 100%;
}

Wichtige Einheiten

pxPixel (fest)
remRelativ zur Root-Schriftgröße
emRelativ zur Eltern-Schriftgröße
%Prozent des Elternelements
vwProzent der Viewport-Breite
vhProzent der Viewport-Höhe
rem ist meistens die beste Wahl für responsive Designs.

Pseudo-Klassen & Responsive

  .container {
    flex-direction: column;
  }
}

JavaScript Cheatsheet

Die Programmiersprache des Webs – Interaktivität & Logik

Variablen

// Veränderbar
let name = "Anna";
name = "Max";    // OK

// Konstante (nicht neu zuweisbar)
const PI = 3.14;
PI = 3;          // Fehler!

// var (veraltet, vermeiden!)
var alt = "nicht nutzen";
Nutze const als Standard, let nur wenn nötig.

Datentypen

String"Hallo" oder 'Hallo'
Number42, 3.14
Booleantrue / false
Array[1, 2, 3]
Object{ name: "Max" }
nullAbsichtlich leer
undefinedNoch kein Wert zugewiesen
// Typ prüfen
typeof "Hallo";  // "string"
typeof 42;       // "number"
typeof true;     // "boolean"

Operatoren

// Arithmetisch
+  -  *  /  %  **

// Vergleich
===  // gleich (strikt)
!==  // ungleich (strikt)
>  <  >=  <=

// Logisch
&&   // UND
||   // ODER
!    // NICHT

// String-Verkettung
"Hallo " + "Welt"
`Hallo ${name}`  // Template Literal
Immer === statt == verwenden!

Bedingungen

if (alter >= 18) {
  console.log("Volljährig");
} else if (alter >= 16) {
  console.log("Fast volljährig");
} else {
  console.log("Minderjährig");
}

// Ternary Operator (Kurzform)
let status = alter >= 18
  ? "Erwachsen"
  : "Minderjährig";

Schleifen

// for-Schleife
for (let i = 0; i < 5; i++) {
  console.log(i);  // 0, 1, 2, 3, 4
}

// for...of (Arrays)
const farben = ["rot", "grün", "blau"];
for (const farbe of farben) {
  console.log(farbe);
}

// while
let count = 0;
while (count < 3) {
  console.log(count);
  count++;
}

Funktionen

// Funktionsdeklaration
function greet(name) {
  return "Hallo " + name;
}

// Arrow Function
const greet = (name) => {
  return "Hallo " + name;
};

// Kurzform (eine Zeile)
const double = (n) => n * 2;

// Aufrufen
greet("Max");    // "Hallo Max"
double(5);       // 10

Arrays & Methoden

const fruits = ["Apfel", "Birne"];

fruits.push("Kirsche");   // Hinzufügen
fruits.pop();             // Letztes entfernen
fruits.length;            // Anzahl
fruits[0];                // "Apfel"
fruits.includes("Birne"); // true

// Iteration
fruits.forEach((fruit) => {
  console.log(fruit);
});

// Transformieren
const upper = fruits.map(
  (f) => f.toUpperCase()
);

// Filtern
const short = fruits.filter(
  (f) => f.length < 6
);

Objekte

const person = {
  name: "Max",
  alter: 25,
  hobbys: ["Lesen", "Coden"]
};

// Zugriff
person.name;          // "Max"
person["alter"];      // 25

// Ändern / Hinzufügen
person.email = "max@mail.de";

// Destructuring
const { name, alter } = person;

String-Methoden

const text = "Hallo Welt";

text.length;          // 10
text.toUpperCase();   // "HALLO WELT"
text.toLowerCase();   // "hallo welt"
text.includes("Welt"); // true
text.indexOf("Welt");  // 6
text.slice(0, 5);      // "Hallo"
text.split(" ");       // ["Hallo","Welt"]
text.trim();           // Leerzeichen weg
text.replace("Welt", "JS");
// "Hallo JS"
